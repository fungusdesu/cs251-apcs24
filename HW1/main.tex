\begin{problem}{2.2.3}
	Which of the following strings are formulas in predicate logic? Specify a reason for failure for strings which aren't, draw parse trees of all strings which are.
	\begin{tasks}[label=\textbf{\alph*.}]
		\task Let $m$ be a constant, $f$ a function symbol with one argument and $S$ and $B$ two predicate symbols, each with two arguments:
		\begin{enumerate}[i.]
			\item $S(m, x)$
			\item $B(m, f(m))$
			\item $f(m)$
			\item $B(B(m, x), y)$
			\item $S(B(m), z)$
			\item $(B(x, y) \rightarrow (\exists z S(z, y)))$
			\item $(S(x, y) \rightarrow S(y, f(f(x))))$
			\item $(B(x) \rightarrow B(B(x)))$
		\end{enumerate}
		\task Let $c$ and $d$ be constants, $f$ a function symbol with one argument, $g$ a function symbol with two arguments and $h$ a function symbol with three arguments. Furthermore, $P$ and $Q$ are predicate symbols with three arguments:
		\begin{enumerate}[i.]
			\item $\forall x P(f(d), h(g(c, x), d, y))$
			\item $\forall x P(f(d), h(P(x, y), d, y))$
			\item $\forall x Q(g(h(x, f(d), x), g(x, x)), h(x, x, x), c)$
			\item $\exists z (Q(z, z, z) \rightarrow P(z))$
			\item $\forall x\forall y (g(x, y) \rightarrow P(x, y, x))$
			\item $Q(c, d, c)$
		\end{enumerate}
	\end{tasks}

The string $S(B(m), z)$ is not a formula, since the string $B(m)$ is not a formula nor term ($B$ is a binary predicate taking in only one argument)

\paragraph{d.}
Performing the substitution $\phi[t/x]$, we yield
\[
	\lnot (\forall x ((\exists y P(x, y, z)) \land (\forall z P(g(f(g(y, y)), y), y, z)))).
\]
Because no variables in the parse tree of $t$ is captured in $\psi$, we have $t$ is free for $x$ in $\psi$.

Performing the substituion $\phi[t/y]$, we yield 
\[
	\lnot (\forall x ((\exists y P(x, y, z)) \land \forall z(P(x, g(f(g(y, y)), y), z)))).
\]
Because no variables in the parse tree of $t$ is captured in $\psi$, we have $t$ is free for $y$ in $\psi$.

Performing the substitution $\phi[t/z]$, we yield
\[
	\lnot (\forall x ((\exists y P(x, y, g(f(g(y, y)), y))) \land (\forall z P(x, y, z)))).
\]
Because the variable $y$ is captured to the symbol $\exists y$, we have $t$ is not free for $z$ in $\psi$.
