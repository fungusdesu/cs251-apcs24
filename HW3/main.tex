\begin{problem}{2.4.2}
	Consider the sentence $\phi \xequals{\text{def}} \forall x \exists y \exists z (P(x, y) \land P(z, y) \land (P(x, z) \rightarrow P(z, x)))$. Which of the following models satisfies $\phi$?
	\begin{tasks}[label=\textbf{\alph*.}]
		\task The model $\mathcal M$ consists of the set of natural numbers with $P^{\mathcal M} \xequals{\text{def}} \{(m, n) \mid m < n\}$.
		\task The model $\mathcal M'$ consists of the set of natural numbers with $P^{\mathcal M'} \xequals{\text{def}} \{(m, 2*m) \mid m \text{ natural number}\}$.
		\task The model $\mathcal M''$ consists of the set of natural numbers with $P^{\mathcal M''} \xequals{\text{def}} \{(m, n) \mid m < n + 1\}$
	\end{tasks}
\end{problem}

\paragraph{a.} Suppose an arbitrary $x$. Choose $y = x + 1$; then it is readily apparent that $P(x, y)$. Choose $z = y - 1$; it is also obvious that $P(z, y)$. Observe that since $z = y - 1 = x$, we obtain $P(x, z)$ is false and thus $P(x, z) \rightarrow P(z, x)$ is true. Therefore, we yield that the model $\mathcal M$ satisfies $\phi$.

\paragraph{b.} Suppose an arbitrary $x$. Choose $y = 2x$ in order to satisfy $P(x, y)$, then $z = x$. Because $P(x, z)$ is false, the implication evaluates to true and thus the model $\mathcal M'$ satisfies $\phi$.

\paragraph{c.} Suppose an arbitrary $x$. Choose $y = x$ to satisfy $P(x, y)$ and choose $z = y$ to satisfy $P(z, y)$. Because $P(x, z)$ and $P(z, x)$ both evaluate to true, the implication does so as well and thus the model $\mathcal M''$ satisfies $\phi$.

\begin{problem}{2.4.3}
	Let $P$ be a predicate with two arguments. Find a model which satisfies the sentence $\forall x \lnot P(x, x)$; also find one which doesn't.
\end{problem}

Define $\mathcal M$ to be the model consisting of the set of natural numbers with $P^{\mathcal M} \xequals{\text{def}} \{(m, n) \mid m < n\}$. This model satisfies the given formula, as $x < x$ is false for any arbitrary $x$. Additionally, define $\mathcal M'$ to be the model consisting of the set of natural numbers with $P^{\mathcal M'} \xequals{\text{def}} \{(m, n) \mid m = n\}$. Because $x = x$ is true for any arbitrary $x$, this model does not safisfy the formula.

\begin{problem}{2.3.7}
	Show the semantic entailment $\forall x \lnot \phi \vDash \lnot\exists x \phi$; for that you have to take any model which satisfies $\forall x \lnot \phi$ and you have to reason why this model must also satisfy $\lnot \exists x \phi$. You should do this in a similar way to the examples in Section 2.4.2.
\end{problem}
\begin{proof}
	Suppose a model $\mathcal M$ satisfying $\forall x \lnot \phi$; we obtain $\mathcal M \vDash_{[x \mapsto a]} \lnot \phi$ holds for all $a$ and thus $\mathcal M \vDash_{[x \mapsto a]} \phi$ holds for none of the $a$. This shows that $\mathcal M \cancel{\vDash}_{[x \mapsto a]} \exists x \phi$ and so we arrive to $\mathcal M \vDash_{[x \mapsto a]} \lnot \exists x \phi$.
\end{proof}

\begin{problem}{2.4.8}
	Show the semantic entailment $\forall x P(x) \lor \forall x Q(x) \vDash \forall x (P(x) \lor Q(x))$.
\end{problem}

\begin{proof}
	Suppose a model $\mathcal M$ satisfying $\forall x P(x) \lor \forall x Q(x)$. Suppose $\mathcal M \vDash_l \forall x P(x)$, then $\mathcal M \vDash_{[x \mapsto a]} P(x)$ for all $a$. Thus one also can conclude $\mathcal M \vDash_{[x \mapsto a]} (P(x) \lor Q(x))$. By symmetry, this also works for the case $\mathcal M \vDash_l \forall x Q(x)$ and thus $\mathcal M \vDash_{[x \mapsto a]} (P(x) \lor Q(x))$. Consequently, $\mathcal M \vDash_l \forall x (P(x) \lor Q(x))$, as desired.
\end{proof}

\begin{problem}{2.4.10}
	Is $\forall x (P(x) \lor Q(x)) \vDash \forall x P(x) \lor \forall x Q(x)$ a semantic entailment? Justify your answer.
\end{problem}

The given string is not a semantic entailment. However, not every element is below $0$ and not every element falls below 0 and so $\mathcal M \vDash \forall x P(x) \lor \forall x Q(x)$ does not hold.

\begin{problem}{2.4.12}

		\task[\textbf{i.}] $(\forall x ((P(x) \rightarrow Q(x)) \land (Q(x) \rightarrow P(x)))) \rightarrow ((\forall x P(x)) \rightarrow (\forall x Q(x)))$.
	\end{tasks}
\end{problem}

For every formula, we apply Gödel's completeness theorem to prove semantic validity via syntactic validity.
\paragraph{a.} For ease of reference, let $\phi$ be the formula
\[
	(\forall x \forall y (S(x, y) \rightarrow S(y, x))) \rightarrow (\forall x \lnot S(x, x)).
\]
Then $\phi$ is not valid. Define by $\mathcal M$ the model consisting of natural numbers in which the predicate $S^{\mathcal M}$ is the equality predicate $=$. The antecedant is clearly true due to the commutativity of equality, yet the consequent is false due to the reflexitivity of equality. Thus, we obtain that $\mathcal M \vDash_l \phi$ does not hold.

\paragraph{c.} For ease of reference, let $\phi$ be the formula

	(\forall x (P(x) \rightarrow \exists y Q(y))) \rightarrow (\forall x \exists y (P(x) \rightarrow Q(y))).
