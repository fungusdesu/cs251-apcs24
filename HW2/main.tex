	\end{subproof}
	\exists x \exists y S(x, y) & $\exists$e 1, 2--9 \qed
\end{logicproof}

\begin{problem}{2.3.14}
	Translate the following argument into a sequent in predicate logic using a suitable set of predicate symbols:
	\begin{quote}
		If there are any tax payers, then all politicians are tax payers. If there are any philanthropists, then all tax payers are philanthropists. So, if there are any tax-paying philanthropists, then all politicians are philanthropists.
	\end{quote}
	Now come up with a proof of that sequent in predicate logic.
\end{problem}

Define the following predicates as follows:
\begin{itemize}
	\item $T(x)$: $x$ is a tax payer
	\item $P(x)$: $x$ is a politician
	\item $H(x)$: $x$ is a philanthropist
\end{itemize}
The given argument can then be translated into the following sequent:
\[
	\exists x T(x) \rightarrow \forall x(P(x) \rightarrow T(x)), \exists x H(x) \rightarrow \forall x (T(x) \rightarrow H(x)) \vdash \exists x (T(x) \land H(x)) \rightarrow \forall x (P(x) \rightarrow H(x)).
\]
This sequent is in fact true. A proof follows.

\newpage
\begin{logicproof}{4}
	\exists x T(x) \rightarrow \forall x(P(x) \rightarrow T(x)) & premise \\
	\exists x H(x) \rightarrow \forall x (T(x) \rightarrow H(x)) & premise \\
	\begin{subproof}
		\exists x (T(x) \land H(x)) & assumption \\
		\begin{subproof}
			x_0 & assumption \\
			T(x_0) \land H(x_0) & assumption \\
			T(x_0) & $\land$e$_1$ 5 \\
			\exists x T(x) & $\exists$i 6 \\
			\forall x (P(x) \rightarrow T(x)) & $\rightarrow$e 1, 7 \\
			H(x_0) & $\land$e$_2$ 5 \\
			\exists x H(x) & $\exists$i 9 \\
			\forall x (T(x) \rightarrow H(x)) & $\rightarrow$e 2, 10 \\
			\begin{subproof}
				\widetilde{x} & assumption \\
				P(\widetilde{x}) \rightarrow T(\widetilde{x}) & $\forall$e 8 \\
				T(\widetilde{x}) \rightarrow H(\widetilde{x}) & $\forall$e 11 \\
				\begin{subproof}
					P(\widetilde{x}) & assumption \\
					T(\widetilde{x}) & $\rightarrow$e 13, 15 \\
					H(\widetilde{x}) & $\rightarrow$e 14, 16
				\end{subproof}
				P(\widetilde{x}) \rightarrow H(\widetilde{x}) & $\rightarrow$i 15--17
			\end{subproof}
			\forall x (P(x) \rightarrow H(x)) & $\forall$i 12--18
		\end{subproof}
		\forall x (P(x) \rightarrow H(x)) & $\exists$e 3, 4--19
	\end{subproof}
	\exists x (T(x) \land H(x)) \rightarrow \forall x (P(x) \rightarrow H(x)) & $\rightarrow$i 3--20 \qed
\end{logicproof}
